//ИГРА КАМЕНЬ НОЖНИЦЫ БУМАГА:
enum tool {ROCK, PAPER, SCISSORS};
enum outcome {P1_WON, P2_WON, DRAW};

enum outcome rps (enum tool p1, enum tool p2) {
	return (enum outcome [3][3]) {
		[ROCK]     = {[ROCK] = DRAW,   [PAPER] = P2_WON, [SCISSORS] = P1_WON},
		[PAPER]    = {[ROCK] = P1_WON, [PAPER] = DRAW,   [SCISSORS] = P2_WON},
		[SCISSORS] = {[ROCK] = P2_WON, [PAPER] = P1_WON, [SCISSORS] = DRAW},
	}[p1][p2];
}
//Это функция, которая принимает два параметра: p1 и p2, оба имеют тип enum tool.
//Функция возвращает значение типа enum outcome.
//Функция использует массив для хранения результатов игры.
//Массив имеет размер 3x3, где каждая строка соответствует одному из значений enum tool,
//а каждый столбец - одному из возможных результатов.
//Например, если p1 равен ROCK и p2 равен PAPER, функция вернет значение P2_WON.
//Если p1 равен PAPER и p2 равен SCISSORS, функция вернет DRAW.И так далее.
